set(CMAKE_SYSTEM_NAME Generic)

if (NOT DEFINED ENV{THEOS_ARCH})
    message(FATAL_ERROR "THEOS_ARCH not set.")
endif()

if (NOT DEFINED ENV{THEOS_ROOT})
    message(FATAL_ERROR "THEOS_ROOT not set.")
endif()

set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -fno-exceptions")

set(SERENITY_BUILD_DIR $ENV{THEOS_ROOT}/Build/$ENV{THEOS_ARCH})

# where to read from/write to
set(CMAKE_SYSROOT ${THEOS_BUILD_DIR}/Root)
set(CMAKE_STAGING_PREFIX ${THEOS_BUILD_DIR}/Root/usr)
set(CMAKE_INSTALL_PREFIX ${THEOS_BUILD_DIR}/Root/usr)
set(CMAKE_INSTALL_DATAROOTDIR ${THEOS_BUILD_DIR}/Root/usr/share)

set(CMAKE_C_COMPILER $ENV{THEOS_ARCH}-pc-serenity-gcc)
set(CMAKE_CXX_COMPILER $ENV{THEOS_ARCH}-pc-serenity-g++)

set(CMAKE_FIND_ROOT_PATH_MODE_PROGRAM NEVER)
set(CMAKE_FIND_ROOT_PATH_MODE_LIBRARY ONLY)
set(CMAKE_FIND_ROOT_PATH_MODE_INCLUDE ONLY)
set(CMAKE_FIND_ROOT_PATH_MODE_PACKAGE ONLY)
